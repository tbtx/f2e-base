overlay 背景遮罩层

参考实例 

    依赖jquery和tbtx.js
    http://a.tbcdn.cn/apps/tbtx/base/js/component/overlay.js


    // 获取overlay实例
    var o = new tbtx.Overlay(option);

    o.show();
    o.hide();


     def = {
            'className': 'overlay', 遮罩层class
            'opacity': 0.5,     透明度
            'color': '#000',    背景颜色
            'zIndex': , // 遮罩层z-index, 可以不传
            'hideOnClick': false, // 点击遮罩层是否关闭
        };
显示和隐藏的监听回调 

    $('body').on('tbtx.overlay.show', function() {
        console.log('show');
    });
    $('body').on('tbtx.overlay.hide', function() {
        console.log('hide');
    });
