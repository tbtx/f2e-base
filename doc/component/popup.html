popup 弹出层


    依赖jquery和tbtx.js
    http://a.tbcdn.cn/apps/tbtx/base/js/component/overlay.popup.js
参考实例 

    var o = new tbtx.Popup(selector, option);
    or
    var o = $(selector).Popup();

    o.show();   // o.show('fadeIn', function) or o.show(function)
    o.hide()       // o.show('fadeOut', function)
已经存在于dom中的元素需要手动remove，通过字符串生成的会在destroy时自动remove 

     var defaults = {
        withOverlay: true,      // 是否使用遮罩层
        overlayOption: {        // 遮罩层的配置，参考overlay

        },
        destoryOnHide: false,   // 隐藏的时候是否删除元素，适用于只需要弹出一次的元素
    };
显示跟隐藏后的回调 

    $('#pop').on('tbtx.popup.show', function() {
        console.log('tbtx.popup.show');
    }).on('tbtx.popup.hide', function() {
        console.log('tbtx.popup.hide');
    });
